#version 460 core //version

layout (location = 0) in vec3 aPos;    //location is in position 0
layout (location = 1) in vec3 aColor;  //color is in position 1

out vec3 vertexColor; //output color of vertex

void main(){
    gl_Position = vec4(aPos.xyz, 1.0); // output vertex position (for other shaders)
    vertexColor = aColor;              // output color (for other shaders)
}
